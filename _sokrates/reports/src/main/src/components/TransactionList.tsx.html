import React, { ReactNode } from "react"; import { styled } from "@mui/material/styles"; import { Paper, Button, ListSubheader, Grid } from "@mui/material"; import { Link as RouterLink } from "react-router-dom"; import { isEmpty } from "lodash/fp"; import SkeletonList from "./SkeletonList"; import { TransactionResponseItem, TransactionPagination } from "../models"; import EmptyList from "./EmptyList"; import TransactionInfiniteList from "./TransactionInfiniteList"; import TransferMoneyIllustration from "./SvgUndrawTransferMoneyRywa"; const PREFIX = "TransactionList"; const classes = { paper: `${PREFIX}-paper`, }; const StyledPaper = styled(Paper)(({ theme }) => ({ [`&.${classes.paper}`]: { paddingLeft: theme.spacing(1), }, })); export interface TransactionListProps { header: string; transactions: TransactionResponseItem[]; isLoading: Boolean; showCreateButton?: Boolean; loadNextPage: Function; pagination: TransactionPagination; filterComponent: ReactNode; } const TransactionList: React.FC<TransactionListProps> = ({ header, transactions, isLoading, showCreateButton, loadNextPage, pagination, filterComponent, }) => { const showEmptyList = !isLoading && transactions?.length === 0; }; export default TransactionList;
